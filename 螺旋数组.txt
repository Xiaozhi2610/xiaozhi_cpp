#include<iostream>
#include <algorithm>
#include <vector>
#include <stdio.h>
using namespace std;

int v[10][10];

int main()
{
	int n = 0;
	int x = 0;
	int y = 0;
	int sum = 1;
	cin >> n;
	int zhuangtai = 0;
	for (int i = 0; i < n * n; i++)
	{
		switch (zhuangtai)
		{
		case 0:
			v[y][x] = sum++;
			x++;
			break;
		case 1:
			v[y][x] = sum++;
			y++;
			break;
		case 2:
			v[y][x] = sum++;
			x--;
			break;
		case 3:
			v[y][x] = sum++;
			y--;
		}
		if (zhuangtai == 0 && (x == n - 1 || v[y][x + 1] != 0))
		{
			zhuangtai = 1;
		}
		if (zhuangtai == 1 && (y == n - 1 || v[y + 1][x] != 0))
		{
			zhuangtai = 2;
		}
		if (zhuangtai == 2 && (x == 0 || v[y][x - 1] != 0))
		{
			zhuangtai = 3;
		}
		if (zhuangtai == 3 && (y == 0 || v[y - 1][x] != 0))
		{
			zhuangtai = 0;
		}
	}






	for (int i = 0; i < n; i++)
	{
		for (int k = 0; k < n; k++)
		{
			printf("%3d", v[i][k]);
		}
		cout << endl;
	}




	return 0;
}
